CREATE OR REPLACE PROCEDURE UPD_ATS_TRNSCTN_LOG_PR (
    P_TRNSCTN_STS_TYP_CDE IN VARCHAR2,
    P_ATS_SNT_DTE IN DATE,
    P_ATS_RPT_ND_FLG IN VARCHAR2,
    P_ERR_MSG_DSC   IN VARCHAR2,
    P_ERR_MSG_TXT IN VARCHAR2,
    P_ATS_TRNSCTN_LOG_ID IN NUMBER,
    P_SUCCESS_MSG IN OUT VARCHAR2)
  IS
  BEGIN
 
  UPDATE ATS_TRNSCTN_LOG SET TRNSCTN_STS_TYP_CDE = P_TRNSCTN_STS_TYP_CDE, ATS_SNT_DTE = P_ATS_SNT_DTE, ATS_RPT_ND_FLG = P_ATS_RPT_ND_FLG,
  ERR_MSG_DSC = P_ERR_MSG_DSC, ERR_MSG_TXT = P_ERR_MSG_TXT,  
  UPDATED_BY_USER_ID = (SELECT USER FROM DUAL) WHERE ATS_TRNSCTN_LOG_ID = P_ATS_TRNSCTN_LOG_ID;
  P_SUCCESS_MSG := 'TRUE';
  COMMIT;
  EXCEPTION
 WHEN OTHERS THEN
   P_SUCCESS_MSG := 'FALSE';
   ROLLBACK;
 
END;
